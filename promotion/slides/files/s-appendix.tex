\begin{frame}{Counterfactual Policy Evaluation}\centering\vspace{-0.5cm}

\begin{columns}
	\begin{column}{0.55\textwidth}
		\begin{itemize}
			\item Counterfactuals to answer "what-if" questions.
			\item Attribute cause and effect between interventions and outcomes.
			\item Especially difficult in scenarios where we can't conduct randomized experiments.
		\end{itemize}

		\medskip
		\heading{Examples}

		\begin{itemize}
			\footnotesize
		\item How would universal basic income on a national scale affect labor supply?
		\item What would the earth climate look like if more climate action was taken?
		\item How do compulsory schooling laws affect educational attainment of youths?
		\end{itemize}
	\end{column}

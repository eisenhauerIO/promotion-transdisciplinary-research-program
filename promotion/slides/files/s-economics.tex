\begin{frame}{Embracing statistical decision theory}\vspace{0.4cm}

  \begin{itemize}\setlength\itemsep{1em}
  \item promote a well-reasoned and transparent policy process
  \item clarify trade-off between choices
  \item examine the coherence of choices
  \item ease ex-post justification
  \item facilitate communication of uncertainty
\end{itemize}\vspace{0.3cm}

\end{frame}
%---------------------------------------------------------------------------------------------------
%---------------------------------------------------------------------------------------------------
\begin{frame}{Decision-theoretic framework}\vspace{0.25cm}


\heading{As-if decisions with point estimates}\vspace{0.3cm}
\begin{itemize}
  \item \only<1-6>{\makebox[0pt][l]{As-if optimization}\phantom{spacespacespacespace}} \only<2-6>{$\quad g^* =\argmax_{g \in \G} U(M_g({\color{red} \hat{\btheta}}))$}
\end{itemize}\vspace{0.4cm}

\pause\pause

\heading{As-if decisions with set estimates}\vspace{0.3cm}
\begin{itemize}
  \item \only<3-6>{\makebox[0pt][l]{Maximin criterion}\phantom{spacespacespacespace}} \only<4-6>{$\quad g^* =\argmax_{g \in \G} \min_{\btheta \in {\color{red} \U(\alpha)}} U(M_g(\btheta))$}

  \item[]
  \item \only<3-6>{\makebox[0pt][l]{Minimax regret rule}\phantom{spacespacespacespace}} \only<5-6>{$\quad g^* =\argmin_{g \in \G} \max_{\btheta \in {\color{red}\U(\alpha)}} \left[\max_{\tilde{g} \in \G} U(M_{\tilde{g}}(\btheta)) - U(M_g(\btheta)) \right]$}

  \item[]
  \item \only<3-6>{\makebox[0pt][l]{Subjective Bayes}\phantom{spacespacespacespace}} \only<6>{$\quad g^* =\argmax_{g \in \G} \int_{ {\color{red} \U(\alpha)}} U(M_g({\btheta})) \dd{f(\btheta)}$}

\end{itemize}

\end{frame}
